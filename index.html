/** * Cloudflare Pages Function — Contact form handler * Path: /api/contact (file lives at /functions/api/contact.js) * * Receives the contact form POST, validates it, and relays it to your * private inbox via the Resend API. The recipient (your Gmail) and the * Resend API key are stored as environment variables / secrets in the * Cloudflare Pages dashboard — they are NEVER exposed to the browser. * * Required environment variables (set in Cloudflare Pages → Settings → * Environment variables): * RESEND_API_KEY — your Resend API key (mark as "Secret/Encrypted") * CONTACT_TO — the Gmail address that should receive submissions * CONTACT_FROM — verified sender, e.g. "Research Site " * (optional; falls back to noreply@ha-capital.org) */ export async function onRequestPost({ request, env }) { const json = (obj, status = 200) => new Response(JSON.stringify(obj), { status, headers: { 'Content-Type': 'application/json' }, }); // --- Parse body ----------------------------------------------------------- let data; try { data = await request.json(); } catch { return json({ error: 'Invalid request body.' }, 400); } const firstName = (data.first_name || '').toString().trim(); const lastName = (data.last_name || '').toString().trim(); const email = (data.email || '').toString().trim(); const topic = (data.topic || '').toString().trim(); const message = (data.message || '').toString().trim(); const honeypot = (data.company || '').toString().trim(); // --- Spam honeypot: real users never fill the hidden "company" field ------ // Return 200 so bots think they succeeded, but send nothing. if (honeypot) { return json({ ok: true }); } // --- Validation ----------------------------------------------------------- const emailOk = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email); if (!firstName || !emailOk || !message) { return json({ error: 'Please provide your name, a valid email, and a message.' }, 400); } if (message.length > 5000) { return json({ error: 'Message is too long.' }, 400); } // --- Config --------------------------------------------------------------- if (!env.RESEND_API_KEY || !env.CONTACT_TO) { return json({ error: 'Server email is not configured.' }, 500); } const fromAddress = env.CONTACT_FROM || 'Research Site '; // --- Build the email ------------------------------------------------------ const fullName = lastName ? `${firstName} ${lastName}` : firstName; const safe = (s) => s.replace(/&/g, '&').replace(//g, '>'); const subject = `Contact form: ${fullName}${topic ? ' — ' + topic : ''}`; const text = `New contact form submission\n\n` + `Name: ${fullName}\n` + `Email: ${email}\n` + `Topic: ${topic || '(none selected)'}\n\n` + `Message:\n${message}\n`; const html = `

New contact form submission

Name: ${safe(fullName)}

Email: ${safe(email)}

Topic: ${safe(topic) || '(none selected)'}

Message:

${safe(message)}

`; // --- Send via Resend ------------------------------------------------------ try { const resp = await fetch('https://api.resend.com/emails', { method: 'POST', headers: { 'Authorization': `Bearer ${env.RESEND_API_KEY}`, 'Content-Type': 'application/json', }, body: JSON.stringify({ from: fromAddress, to: [env.CONTACT_TO], reply_to: email, // hitting "Reply" in Gmail goes to the visitor subject, text, html, }), }); if (!resp.ok) { const detail = await resp.text().catch(() => ''); console.error('Resend error:', resp.status, detail); return json({ error: 'Email service rejected the request.' }, 502); } return json({ ok: true }); } catch (err) { console.error('Send failed:', err); return json({ error: 'Failed to send message.' }, 500); } } // Optional: reject non-POST methods cleanly. export async function onRequest({ request }) { if (request.method !== 'POST') { return new Response('Method Not Allowed', { status: 405 }); } }
